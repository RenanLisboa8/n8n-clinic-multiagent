## System Monitoring Dashboard (Sample Queries)

```sql
-- ğŸ“Š DAILY DASHBOARD --

-- 1. Cache Hit Rate
SELECT 
  ROUND(
    COUNT(*) FILTER (WHERE source = 'cache') * 100.0 / COUNT(*), 
    2
  ) as cache_hit_rate_pct
FROM message_logs
WHERE created_at > NOW() - INTERVAL '24 hours';

-- 2. Average Response Times
SELECT 
  source,
  ROUND(AVG(response_time_ms)) as avg_ms,
  ROUND(PERCENTILE_CONT(0.95) WITHIN GROUP (ORDER BY response_time_ms)) as p95_ms
FROM message_logs
WHERE created_at > NOW() - INTERVAL '24 hours'
GROUP BY source;

-- 3. Cost Tracking
SELECT 
  t.clinic_name,
  COUNT(*) as total_messages,
  COUNT(*) FILTER (WHERE ai_called = true) as ai_calls,
  ROUND(COUNT(*) FILTER (WHERE ai_called = true) * 0.012, 2) as cost_usd
FROM message_logs m
JOIN tenant_config t ON m.tenant_id = t.tenant_id
WHERE m.created_at > NOW() - INTERVAL '24 hours'
GROUP BY t.clinic_name;

-- 4. Top FAQs
SELECT 
  question_original,
  view_count,
  ROUND(view_count * 100.0 / SUM(view_count) OVER(), 2) as pct
FROM tenant_faq
WHERE tenant_id = 'YOUR_TENANT_ID'
ORDER BY view_count DESC
LIMIT 10;
```
